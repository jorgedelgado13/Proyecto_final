# ====== Funci√≥n de limpieza (conserva '√±', quita tildes) ======
# En lugar de NFD (que romper√≠a la √±), mapeamos tildes expl√≠citamente.
_TILDE_MAP = str.maketrans({
    "√°": "a", "√©": "e", "√≠": "i", "√≥": "o", "√∫": "u",
    "√Å": "a", "√â": "e", "√ç": "i", "√ì": "o", "√ö": "u",
    "√º": "u", "√ú": "u"
})

def normalize_text(text: str) -> str:
    text = str(text).lower().strip()
    # URLs
    text = re.sub(r"https?://\S+|www\.\S+", " ", text)
    # menciones
    text = re.sub(r"@\w+", " ", text)
    # hashtags (si no te sirven)
    text = re.sub(r"#\w+", " ", text)
    # quitar tildes pero NO tocar '√±'
    text = text.translate(_TILDE_MAP)
    # conservar letras (a-z, √±) y espacios
    text = re.sub(r"[^a-z√±\s]", " ", text)
    # colapsar espacios
    text = re.sub(r"\s+", " ", text).strip()
    return text
